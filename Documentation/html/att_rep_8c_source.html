328 // gets the metric distance between two points given in the LlaCoor_f format
329 static float getDistance(struct LlaCoor_f *own_pos, struct LlaCoor_f *goal_pos)
330 {
331  return GLOBE_RADIUS * acos(
332  sin(own_pos->lat) * sin(goal_pos->lat) +
333  cos(own_pos->lat) * cos(goal_pos->lat) *
334  cos(own_pos->lon - goal_pos->lon));
335 }

361 // attraction_force = ka/max(distance,0.0001) * dist_vec
362 static struct EnuCoor_f attract(struct EnuCoor_f *own_pos, struct EnuCoor_f *pos_ac, struct EnuCoor_f *acc, float multiplier)
363 {
364  struct EnuCoor_f force = {
365  pos_ac->x - own_pos->x,
366  pos_ac->y - own_pos->y,
367  0.0f};
368  msg.attraction_force = force;
369 
370  float d = sqrtf(force.x * force.x + force.y * force.y);
371  msg.attraction_d = d;
372  msg.attraction = true;
373 
374  d = fmaxf(0.0001f, d);
375  float strength = GRAVITY * multiplier / d;
376  msg.attraction_strength = strength;
377 
378  force.x = force.x * strength;
379  force.y = force.y * strength;
380  msg.attraction_force = force;
381 
382  acc->x += force.x;
383  acc->y += force.y;
384 
385  return force;
386 }

388 // repulsion_force = (kb*exp(-distance/2r)) * dist_vec
389 static struct EnuCoor_f repulse(struct EnuCoor_f *own_pos, struct EnuCoor_f *pos_ac, struct EnuCoor_f *acc, float perlimiter, float multiplier)
390 {
391  struct EnuCoor_f force = {
392  pos_ac->x - own_pos->x,
393  pos_ac->y - own_pos->y,
394  0.0f};
395  msg.repulsion_force = force;
396 
397  float d = sqrtf(force.x * force.x + force.y * force.y);
398  msg.repulsion_d = d;
399  msg.repulsion = true;
400 
401  float strength = (GRAVITY * multiplier) * expf(-(d * d) / (2 * perlimiter * perlimiter));
402  msg.repulsion_strength = strength;
403 
404  force.x = force.x * strength;
405  force.y = force.y * strength;
406  msg.repulsion_force = force;
407 
408  acc->x -= force.x;
409  acc->y -= force.y;
410 
411  return force;
412 }

414 // total_force = (kad*max(distance-d,0.0)/max(distance,1.0) - kbd*exp(-distance/(dr/log(kbd/kad)))) * dist_vec
415 static struct EnuCoor_f attRep(struct EnuCoor_f *own_pos, struct EnuCoor_f *pos_ac, struct EnuCoor_f *acc, float reg_size, float comfy_dist, float att_multiplier, float rep_multiplier)
416 {
417  struct EnuCoor_f force = {
418  pos_ac->x - own_pos->x,
419  pos_ac->y - own_pos->y,
420  0.0f};
421  msg.repulsion_force = force;
422  msg.attraction_force = force;
423 
424  float d = sqrtf(force.x * force.x + force.y * force.y);
425  float c = (reg_size * reg_size) / logf(rep_multiplier / att_multiplier);
426  msg.repulsion_d = d;
427  msg.attraction_d = d;
428  msg.repulsion = true;
429  msg.attraction = true;
430 
431  float strength_att = (GRAVITY * att_multiplier) * (fmax(d - comfy_dist, 0.0f) / fmax(d, 1.0f));
432  float strength_rep = (GRAVITY * rep_multiplier) * expf(-(d * d) / c);
433  msg.attraction_strength = strength_att;
434  msg.repulsion_strength = strength_rep;
435 
436  force.x = force.x * (strength_att - strength_rep);
437  force.y = force.y * (strength_att - strength_rep);
438  msg.attraction_force = force;
439  msg.repulsion_force = force;
440 
441  acc->x += force.x;
442  acc->y += force.y;
443 
444  return force;
445 }

448  * swarm_follow_wp(void)
449  * updates the FOLLOW_WAYPOINT_ID to a fixed offset from the last received location
450  * of other aircraft with id FOLLOW_AC_ID
451  */
452 void swarm_follow_wp(void)
453 {
454  struct EnuCoor_f att_force = {0.0f, 0.0f, 0.0f};
455  struct EnuCoor_f member_force = {0.0f, 0.0f, 0.0f};
456  struct EnuCoor_f rep_force = {0.0f, 0.0f, 0.0f};
457  setPoint(&acc, 0.0f, 0.0f, 0.0f);
458 
459  struct EnuCoor_f *own_pos = stateGetPositionEnu_f();
460  msg.own_pos = *own_pos;
461 
462  for (uint8_t ac_id = FIRST_SWARM_MEMBER_ID; ac_id <= LAST_SWARM_MEMBER_ID; ++ac_id)
463  {
464  struct EnuCoor_f *ac_pos = getPositionEnu_f(ac_id);
465  if (ac_pos != NULL && ac_id != AC_ID)
466  {
467  msg.target_pos = *ac_pos;
468  msg.target_ac_id = ac_id;
469  addToPoint(&member_force, attRep(own_pos, ac_pos, &acc, DRONE_REGION_SIZE, COMFY_DIST, DRONE_ATTRECTION_MULTIPLIER, DRONE_REPULSION_MULTIPLIER));
470  }
471  else
472  {
473  msg.target_ac_id = 0;
474  msg.attraction = false;
475  msg.repulsion = false;
476  }
477  }
478 
479  for (uint8_t wp_id = FIRST_ATTRACTION_POINT_ID; wp_id < LAST_ATTRACTION_POINT_ID; ++wp_id)
480  {
481  struct EnuCoor_f current_att_point = {0.0f, 0.0f, 0.0f};
482  current_att_point.x = waypoint_get_x(wp_id);
483  current_att_point.y = waypoint_get_y(wp_id);
484  if (current_att_point.x != 0.0f && current_att_point.y != 0.0f)
485  addToPoint(&att_force, attract(own_pos, &current_att_point, &acc, ATTRECTION_MULTIPLIER));
486  }
487 
488  for (uint8_t wp_id = FIRST_REPELL_POINT_ID; wp_id < LAST_REPELL_POINT_ID; ++wp_id)
489  {
490  struct EnuCoor_f current_rep_point = {0.0f, 0.0f, 0.0f};
491  current_rep_point.x = waypoint_get_x(wp_id);
492  current_rep_point.y = waypoint_get_y(wp_id);
493  if (current_rep_point.x != 0.0f && current_rep_point.y != 0.0f)
494  addToPoint(&rep_force, repulse(own_pos, &current_rep_point, &acc, REGION_SIZE, REPULSION_MULTIPLIER));
495  }
496 
497  struct EnuCoor_f *vel = acInfoGetVelocityEnu_f(AC_ID);
498  vel->x = VELOCITY_LIMIT * tanhf(vel->x + acc.x);
499  vel->y = VELOCITY_LIMIT * tanhf(vel->y + acc.y);
500  acInfoSetVelocityEnu_f(AC_ID, vel);
501 
502  struct EnuCoor_i future_pos = *stateGetPositionEnu_i();
503  future_pos.x += POS_BFP_OF_REAL(vel->x);
504  future_pos.y += POS_BFP_OF_REAL(vel->y);
505  future_pos.z = POS_BFP_OF_REAL(FLIGHT_HEIGHT);
506 
507  if (ticks == 0)
508  {
509  ticks = 3;
510  write_to_log_file(&future_pos, vel, &acc);
511  write_to_debug_file(&future_pos, vel, &acc, &att_force, &member_force, &rep_force);
512  }
513  else
514  --ticks;
515 
516  // Move the waypoints
517  current_pos = *stateGetPositionLla_f();
518  waypoint_set_enu_i(SWARM_WAYPOINT_ID, &future_pos);
519  for (uint8_t wp_id = FIRST_ATTRACTION_POINT_ID; wp_id < LAST_ATTRACTION_POINT_ID; ++wp_id)
520  {
521  struct LlaCoor_i *way_point = waypoint_get_lla(wp_id);
522  att_point = toFloatPointFormat(way_point);
523  updateSyncLinkMsg(&current_pos, wp_id);
524  }
525 }
